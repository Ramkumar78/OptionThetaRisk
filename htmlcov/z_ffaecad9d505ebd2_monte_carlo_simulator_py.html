8class MonteCarloSimulator: 

9 def __init__(self, trades: list, initial_capital: float = 10000.0): 

10 """ 

11 :param trades: List of dicts, each containing 'return_pct'. 

12 :param initial_capital: Starting equity for simulations. 

13 """ 

14 self.trades = trades 

15 self.initial_capital = initial_capital 

16 # Extract percentage returns (e.g., 5.0 for 5%) 

17 self.returns_pct = [t.get('return_pct', 0.0) for t in trades if t.get('return_pct') is not None] 

18 

19 def run(self, simulations: int = 10000, ruin_threshold_pct: float = RUIN_THRESHOLD_PCT): 

20 if not self.returns_pct: 

21 return {"error": "No trade returns to simulate."} 

22 

23 n_trades = len(self.returns_pct) 

24 if n_trades < 10: 

25 return {"error": "Not enough trades for meaningful Monte Carlo (min 10)."} 

26 

27 # Convert to decimal returns for calculation (e.g., 0.05) 

28 returns_decimal = np.array(self.returns_pct) / 100.0 

29 

30 # Pre-generate all random indices at once 

31 # Shape: (simulations, n_trades) 

32 rng = np.random.default_rng() 

33 random_indices = rng.integers(0, n_trades, size=(simulations, n_trades)) 

34 

35 # Select returns: (simulations, n_trades) 

36 # This creates a copy 

37 sim_returns = returns_decimal[random_indices] 

38 

39 # Calculate Equity Curves 

40 # Start with initial_capital, then multiply by (1 + r) 

41 # Pre-allocate full array to avoid copy during hstack 

42 # Shape: (simulations, n_trades + 1) 

43 equity_curves = np.empty((simulations, n_trades + 1), dtype=np.float64) 

44 equity_curves[:, 0] = self.initial_capital 

45 

46 # Calculate cumulative returns 

47 # sim_returns becomes (1 + r) 

48 np.add(sim_returns, 1.0, out=sim_returns) 

49 

50 # Calculate cumulative product into the allocated buffer (columns 1 to end) 

51 np.cumprod(sim_returns, axis=1, out=equity_curves[:, 1:]) 

52 

53 # Scale by initial capital (in-place) 

54 np.multiply(equity_curves[:, 1:], self.initial_capital, out=equity_curves[:, 1:]) 

55 

56 # Calculate Percentiles of Equity Curves ONCE 

57 # Axis 0 is simulations, Axis 1 is steps 

58 percentiles = [5, 25, 50, 75, 95] 

59 # shape: (5, n_trades + 1) 

60 equity_quantiles = np.percentile(equity_curves, percentiles, axis=0) 

61 

62 # Sample Curves 

63 # If simulations > 5, take 5 random. Else take all. 

64 n_samples = min(simulations, 5) 

65 sample_indices = rng.choice(simulations, n_samples, replace=False) 

66 sample_curves = equity_curves[sample_indices] 

67 

68 # Max Drawdown Calculation 

69 # Running Max 

70 running_max = np.maximum.accumulate(equity_curves, axis=1) 

71 # Drawdown at each point: (equity - peak) / peak = (equity/peak) - 1 

72 drawdowns = (equity_curves / running_max) - 1.0 

73 

74 # Max Drawdown for each simulation (min value since drawdowns are negative or 0) 

75 max_dds = np.min(drawdowns, axis=1) # e.g., -0.20 for 20% DD 

76 

77 # Risk of Ruin (> 50% Drawdown) 

78 ruin_mask = max_dds < -0.50 

79 prob_ruin = (np.sum(ruin_mask) / simulations) * 100.0 

80 # Risk of Ruin (Equity < 0 or DD > 50%? Usually Ruin is losing everything or hitting a hard stop) 

81 # Let's define Ruin as hitting < 50% of starting capital (severe) or 0 

82 # Common def: Ruin is blowing up account. Let's say < 0. 

83 # But in pure compounding, it never hits < 0 unless return is <= -100%. 

84 # If returns are simple PnL added, it can go < 0. Backtester uses simple compounding logic? 

85 # UnifiedBacktester: self.equity -= (shares * price); self.equity += proceeds. 

86 # It buys shares. If price goes to 0, you lose 100% of trade. 

87 # So return is -1. 1 + (-1) = 0. Equity becomes 0. 

88 

89 # Let's define Ruin as > ruin_threshold_pct Drawdown 

90 ruin_mask = max_dds < -ruin_threshold_pct 

91 ruin_count = np.sum(ruin_mask) 

92 

93 # Stats from Percentiles (last column corresponds to final equity distribution) 

94 # 0: p5, 1: p25, 2: p50 (median), 3: p75, 4: p95 

95 pct5_final = equity_quantiles[0, -1] 

96 median_final = equity_quantiles[2, -1] 

97 pct95_final = equity_quantiles[4, -1] 

98 

99 # Median Drawdown & Worst Case Drawdown 

100 median_dd = np.median(max_dds) * 100 

101 pct95_dd = np.percentile(max_dds, 5) * 100 

102 

103 # Expected Return (Average) - Mean is separate from percentiles 

104 avg_final_equity = np.mean(equity_curves[:, -1]) 

105 avg_return_pct = ((avg_final_equity - self.initial_capital) / self.initial_capital) * 100 

106 

107 worst_case_return = ((pct5_final - self.initial_capital)/self.initial_capital)*100 

108 best_case_return = ((pct95_final - self.initial_capital)/self.initial_capital)*100 

109 

110 # Extract percentile lists once 

111 p05_list = np.round(equity_quantiles[0], 2).tolist() 

112 p25_list = np.round(equity_quantiles[1], 2).tolist() 

113 p50_list = np.round(equity_quantiles[2], 2).tolist() 

114 p75_list = np.round(equity_quantiles[3], 2).tolist() 

115 p95_list = np.round(equity_quantiles[4], 2).tolist() 

116 

117 curves_data = { 

118 "p05": p05_list, 

119 "p25": p25_list, 

120 "p50": p50_list, 

121 "p75": p75_list, 

122 "p95": p95_list, 

123 } 

124 

125 return { 

126 "simulations": simulations, 

127 "initial_capital": self.initial_capital, 

128 "prob_ruin_50pct": round(prob_ruin, 2), # % chance of 50% drawdown 

129 "median_final_equity": round(median_final, 2), 

130 "avg_return_pct": round(avg_return_pct, 2), 

131 "worst_case_return": round(worst_case_return, 2), 

132 "best_case_return": round(best_case_return, 2), 

133 "median_drawdown": round(median_dd, 2), 

134 "worst_case_drawdown": round(pct95_dd, 2), # 95% Confidence Level Max Drawdown 

135 "equity_curve_percentiles": { 

136 "p5": p05_list, 

137 "p25": p25_list, 

138 "p50": p50_list, 

139 "p75": p75_list, 

140 "p95": p95_list, 

141 }, 

142 "sample_equity_curves": sample_curves.tolist(), 

143 "equity_curves": curves_data, 

144 "message": f"Ran {simulations} simulations. {round(prob_ruin, 2)}% risk of >{int(ruin_threshold_pct * 100)}% drawdown." 

145 }

147def run_simple_monte_carlo(strategies: List[Any], start_equity: float, num_sims: int = 1000, forecast_trades: int = 50, ruin_threshold_pct: float = RUIN_THRESHOLD_PCT) -> Dict: 

148 """ 

149 Runs a Monte Carlo simulation to project future portfolio performance. 

150 Returns: Probability of Ruin, Median Outcome, and 5th Percentile (Worst Case). 

151 """ 

152 if not strategies or len(strategies) < 10: 

153 return {"error": "Need at least 10 trades for Monte Carlo"} 

154 

155 # 1. Extract Trade Statistics 

156 # We use the actual PnL distribution, not just averages, to capture "fat tails" (outliers) 

157 pnls = [s.net_pnl for s in strategies] 

158 

159 # 2. Run Simulations (Vectorized with NumPy for speed) 

160 # We simulate 'forecast_trades' into the future, 'num_sims' times. 

161 # We randomly sample from your HISTORICAL PnL list. 

162 # This assumes your future performance distribution matches your past. 

163 

164 # Shape: (num_sims, forecast_trades) 

165 simulated_trades = np.random.choice(pnls, size=(num_sims, forecast_trades), replace=True) 

166 

167 # Cumulative Sum to get Equity Curve 

168 # Axis 1 = across the trades in a single simulation 

169 sim_curves = np.cumsum(simulated_trades, axis=1) + start_equity 

170 

171 # 3. Calculate Metrics 

172 

173 # Ending Equity for all simulations 

174 ending_equities = sim_curves[:, -1] 

175 

176 # Risk of Ruin (Probability that equity drops below 50% of start at ANY point) 

177 # Check if ANY point in the curve < start_equity * (1 - ruin_threshold_pct) 

178 ruin_threshold = start_equity * (1 - ruin_threshold_pct) 

179 min_equities = np.min(sim_curves, axis=1) 

180 ruin_count = np.sum(min_equities < ruin_threshold) 

181 risk_of_ruin_pct = (ruin_count / num_sims) * 100.0 

182 

183 # Percentiles 

184 median_outcome = np.percentile(ending_equities, 50) 

185 worst_case_outcome = np.percentile(ending_equities, 5) # 5th percentile (95% confidence you won't do worse) 

186 best_case_outcome = np.percentile(ending_equities, 95) 

187 

188 return { 

189 "simulations": num_sims, 

190 "forecast_trades": forecast_trades, 

191 "risk_of_ruin_50pct": round(float(risk_of_ruin_pct), 1), 

192 "median_equity": round(float(median_outcome), 2), 

193 "worst_case_equity": round(float(worst_case_outcome), 2), 

194 "best_case_equity": round(float(best_case_outcome), 2), 

195 "expected_profit": round(float(median_outcome - start_equity), 2) 

196 }
