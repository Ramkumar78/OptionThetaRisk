41def apply_currency_conversion(df: pd.DataFrame, target_currency: str = 'GBP', source_currency: str = 'GBp') -> pd.DataFrame: 

42 """ 

43 Converts DataFrame OHLCV data to target currency. 

44 

45 :param df: The DataFrame containing market data. 

46 :param target_currency: The desired currency (GBP, USD, EUR, etc). 

47 :param source_currency: The source currency. 

48 Defaults to 'GBp' (Pence) which is standard for LSE (.L) tickers. 

49 For Euro tickers, caller should pass 'EUR'. 

50 """ 

51 if df.empty: 

52 return df 

53 

54 # Normalize GBp (pence) to GBP (pounds) first if needed 

55 is_pence = (source_currency == 'GBp') 

56 normalization_factor = 0.01 if is_pence else 1.0 

57 actual_source_iso = 'GBP' if is_pence else source_currency 

58 

59 # Optimization: If target is GBP and source is GBp, just scale by 0.01 

60 # If target is GBp and source is GBp, do nothing. 

61 if source_currency == target_currency: 

62 return df 

63 

64 # Fetch rate from Actual Source (e.g. GBP) to Target (e.g. USD) 

65 try: 

66 exchange_rate = fetch_exchange_rate(actual_source_iso, target_currency) 

67 except Exception as e: 

68 logger.error(f"Failed to fetch exchange rate: {e}") 

69 return df 

70 

71 # Total multiplier = Normalization * Exchange Rate 

72 # e.g. GBp -> USD: 0.01 * (GBP->USD Rate) 

73 # e.g. GBp -> GBP: 0.01 * 1.0 = 0.01 

74 final_multiplier = normalization_factor * exchange_rate 

75 

76 if final_multiplier == 1.0: 

77 return df 

78 

79 df_conv = df.copy() 

80 cols_to_convert = ['Open', 'High', 'Low', 'Close', 'Adj Close'] 

81 

82 logger.info(f"Converting UK Data from {source_currency} to {target_currency} (Multiplier: {final_multiplier})") 

83 

84 for col in cols_to_convert: 

85 if col in df_conv.columns: 

86 df_conv[col] = df_conv[col] * final_multiplier 

87 

88 return df_conv
