11def check_itm_risk(open_groups: List[TradeGroup], prices: Dict[str, float]) -> Tuple[bool, float, List[str]]: 

12 """ 

13 Checks Net Intrinsic Risk per symbol to handle Spreads and Covered Calls correctly. 

14 Returns (is_risky_flag, total_itm_amount, list_of_risk_descriptions). 

15 """ 

16 risky = False 

17 total_net_exposure = 0.0 

18 details = [] 

19 

20 # 1. Group positions by Symbol 

21 by_symbol = defaultdict(list) 

22 for g in open_groups: 

23 if g.symbol in prices: 

24 by_symbol[g.symbol].append(g) 

25 

26 # 2. Analyze Net Risk per Symbol 

27 for symbol, groups in by_symbol.items(): 

28 current_price = prices[symbol] 

29 net_intrinsic_val = 0.0 

30 

31 # Calculate the Net Liquidation Value of options if expired NOW 

32 for g in groups: 

33 qty = g.qty_net 

34 

35 # Stock (Treat as asset/liability at full market value) 

36 if not g.strike or g.right not in ['C', 'P']: 

37 net_intrinsic_val += current_price * qty 

38 continue 

39 

40 if g.strike and g.right: 

41 intrinsic = 0.0 

42 if g.right == 'C': # Call 

43 # Value = Max(0, Price - Strike) 

44 val = max(0.0, current_price - g.strike) 

45 intrinsic = val * qty * 100 

46 elif g.right == 'P': # Put 

47 # Value = Max(0, Strike - Price) 

48 val = max(0.0, g.strike - current_price) 

49 intrinsic = val * qty * 100 

50 

51 # Add to net total (Longs add value, Shorts subtract value) 

52 net_intrinsic_val += intrinsic 

53 

54 # 3. Verdict on this Symbol 

55 # If the Net Intrinsic Value is significantly negative (e.g. < -$500), 

56 # it means the Short legs are ITM and NOT fully covered by Long legs. 

57 if net_intrinsic_val < -500: 

58 risky = True 

59 total_net_exposure += abs(net_intrinsic_val) 

60 details.append(f"{symbol}: Net ITM Exposure -${abs(net_intrinsic_val):,.0f} (Unhedged)") 

61 

62 return risky, total_net_exposure, details

64def calculate_black_swan_impact(open_groups: List[TradeGroup], prices: Dict[str, float]) -> List[StressTestResult]: 

65 """ 

66 Calculates portfolio PnL impact under Black Swan scenarios (+/- 5%, 10%, 20%). 

67 Uses Theoretical Black-Scholes pricing with constant volatility assumption. 

68 """ 

69 results = [] 

70 scenarios = [ 

71 ("Market -20%", -0.20), 

72 ("Market -10%", -0.10), 

73 ("Market -5%", -0.05), 

74 ("Market +5%", 0.05), 

75 ("Market +10%", 0.10), 

76 ("Market +20%", 0.20) 

77 ] 

78 

79 now = pd.Timestamp.now() 

80 # Default Assumptions if data missing 

81 risk_free_rate = 0.045 

82 default_vol = 0.40 

83 

84 for name, move_pct in scenarios: 

85 total_current_val = 0.0 

86 total_new_val = 0.0 

87 

88 # We need to calculate portfolio value at current prices vs new prices 

89 # Iterate all groups 

90 for g in open_groups: 

91 # Handle both object and dict (if serialized) 

92 if isinstance(g, dict): 

93 symbol = g.get("symbol") 

94 qty = g.get("qty_net", g.get("qty_open", 0.0)) 

95 strike = g.get("strike") 

96 otype = g.get("right") 

97 # contract string fallback? 'P 400.0' 

98 if not otype and g.get("contract") and " " in g.get("contract", ""): 

99 parts = g["contract"].split(" ") 

100 otype = parts[0] 

101 if not strike: 

102 try: 

103 strike = float(parts[1]) 

104 except: pass 

105 

106 expiry = g.get("expiry") 

107 else: 

108 symbol = g.symbol 

109 qty = g.qty_net 

110 strike = g.strike 

111 otype = g.right 

112 expiry = g.expiry 

113 

114 if symbol not in prices: 

115 continue 

116 

117 S_curr = prices[symbol] 

118 S_new = S_curr * (1 + move_pct) 

119 

120 # Check if Option or Stock 

121 is_option = strike is not None and otype in ['C', 'P'] 

122 

123 if not is_option: 

124 # Stock Logic 

125 val_curr = qty * S_curr 

126 val_new = qty * S_new 

127 total_current_val += val_curr 

128 total_new_val += val_new 

129 else: 

130 # Option Logic 

131 K = strike 

132 # otype is already extracted 

133 

134 # Calculate T (Time to Expiry) 

135 T = 0.0 

136 if expiry: 

137 try: 

138 # ensure expiry is timestamp 

139 exp_ts = pd.to_datetime(expiry) 

140 # Time to expiry in years 

141 # If expired, T=0 

142 diff = (exp_ts - now).total_seconds() 

143 if diff > 0: 

144 T = diff / (365.0 * 24 * 3600) 

145 except Exception: 

146 T = 0.0 

147 

148 # Map 'C' -> 'call', 'P' -> 'put' 

149 otype_param = 'call' if otype == 'C' else 'put' if otype == 'P' else 'call' 

150 

151 # Calculate Option Prices 

152 price_curr = calculate_option_price(S_curr, K, T, risk_free_rate, default_vol, otype_param) 

153 price_new = calculate_option_price(S_new, K, T, risk_free_rate, default_vol, otype_param) # Constant Vol assumption 

154 

155 val_curr = price_curr * 100 * qty 

156 val_new = price_new * 100 * qty 

157 

158 total_current_val += val_curr 

159 total_new_val += val_new 

160 

161 pnl = total_new_val - total_current_val 

162 pnl_pct = (pnl / abs(total_current_val)) * 100 if total_current_val != 0 else 0.0 

163 

164 results.append(StressTestResult( 

165 scenario_name=name, 

166 market_move_pct=move_pct * 100, 

167 portfolio_value_change=pnl, 

168 portfolio_value_change_pct=pnl_pct, 

169 details=[] 

170 )) 

171 

172 return results

174def calculate_discipline_score(strategies: List[Any], open_positions: List[Dict]) -> Tuple[int, List[str]]: 

175 """ 

176 Calculates a 'Trader Discipline Score' (0-100) based on behavioral metrics. 

177 """ 

178 score = 100 

179 details = [] 

180 

181 # 1. Strategy Analysis (History) 

182 # Penalize Revenge Trading 

183 revenge_count = sum(1 for s in strategies if getattr(s, "is_revenge", False)) 

184 if revenge_count > 0: 

185 penalty = revenge_count * 10 

186 score -= penalty 

187 details.append(f"Revenge Trading: -{penalty} pts ({revenge_count} instances)") 

188 

189 # Reward Early Loss Cutting 

190 # Definition: Loss trade closed faster than average win/loss hold time? 

191 # Simple proxy: Loss < average hold time * 0.5 

192 avg_hold = 0.0 

193 if strategies: 

194 avg_hold = np.mean([s.hold_days() for s in strategies]) 

195 

196 early_cuts = 0 

197 for s in strategies: 

198 if s.net_pnl < 0 and s.hold_days() < (avg_hold * 0.5): 

199 early_cuts += 1 

200 

201 if early_cuts > 0: 

202 bonus = min(20, early_cuts * 2) # Cap bonus 

203 score += bonus 

204 details.append(f"Cutting Losses Early: +{bonus} pts") 

205 

206 # Penalize Tilt (Rapid losses) 

207 losses = [s for s in strategies if s.net_pnl < 0 and getattr(s, 'exit_ts', None)] 

208 losses.sort(key=lambda x: x.exit_ts) 

209 

210 tilt_events = 0 

211 for i in range(len(losses)): 

212 current = losses[i] 

213 window_start = current.exit_ts - pd.Timedelta(hours=24) 

214 

215 count = 0 

216 for j in range(i, -1, -1): 

217 if losses[j].exit_ts < window_start: 

218 break 

219 count += 1 

220 

221 if count > 3: 

222 tilt_events += 1 

223 

224 if tilt_events > 0: 

225 p = tilt_events * 5 

226 score -= p 

227 details.append(f"Tilt Detected (>3 losses in 24h): -{p} pts") 

228 

229 # Reward Patience (Holding Winners) 

230 strategy_hold_avgs = defaultdict(list) 

231 for s in strategies: 

232 name = getattr(s, 'strategy_name', "Unclassified") 

233 strategy_hold_avgs[name].append(s.hold_days()) 

234 

235 avg_map = {k: np.mean(v) for k, v in strategy_hold_avgs.items()} 

236 

237 patience_count = 0 

238 for s in strategies: 

239 name = getattr(s, 'strategy_name', "Unclassified") 

240 if s.net_pnl > 0: 

241 avg = avg_map.get(name, 0) 

242 if avg > 0 and s.hold_days() > avg: 

243 patience_count += 1 

244 

245 if patience_count > 0: 

246 b = min(20, patience_count * 2) 

247 score += b 

248 details.append(f"Patience Bonus (Holding Winners): +{b} pts") 

249 

250 # 2. Open Position Analysis (Risk Management) 

251 # Penalize Gamma Risk (holding < 3 DTE) 

252 gamma_risks = 0 

253 for p in open_positions: 

254 dte = p.get("dte") 

255 # Ensure DTE is valid number 

256 if dte is not None and isinstance(dte, (int, float)) and dte <= 3: 

257 gamma_risks += 1 

258 

259 if gamma_risks > 0: 

260 penalty = gamma_risks * 5 

261 score -= penalty 

262 details.append(f"Gamma Risk (Held < 3 DTE): -{penalty} pts") 

263 

264 # Clamp Score 

265 score = max(0, min(100, score)) 

266 

267 return score, details

269def calculate_kelly_criterion(win_rate: float, profit_factor: float) -> float: 

270 """ 

271 Calculates the Kelly Criterion percentage for position sizing. 

272 

273 Formula derived from standard Kelly f = p - q/b, where: 

274 p = win_rate 

275 b = AvgWin/AvgLoss 

276 Profit Factor (PF) = (p * AvgWin) / ((1-p) * AvgLoss) => b = PF * (1-p)/p 

277 

278 Resulting simplified formula: Kelly % = Win Rate * (1 - 1 / Profit Factor) 

279 

280 Args: 

281 win_rate: The probability of winning (0.0 to 1.0). 

282 profit_factor: The ratio of Gross Profit / Gross Loss. 

283 

284 Returns: 

285 The optimal fraction of the bankroll to wager (0.0 to 1.0). 

286 Returns 0.0 if the Profit Factor is <= 1 or calculation fails. 

287 """ 

288 if profit_factor <= 1.0 or win_rate <= 0.0 or win_rate > 1.0: 

289 return 0.0 

290 

291 kelly_pct = win_rate * (1.0 - 1.0 / profit_factor) 

292 

293 return max(0.0, kelly_pct)
